## 1. Python Example

```python
# A class to represent a simple bank account
class BankAccount:
    def __init__(self, owner: str, balance: float = 0.0):
        self.owner = owner
        self.balance = balance

    def deposit(self, amount: float) -> None:
        if amount <= 0:
            raise ValueError("Deposit must be positive")
        self.balance += amount

    def withdraw(self, amount: float) -> None:
        if amount > self.balance:
            raise ValueError("Insufficient funds")
        self.balance -= amount

    def __str__(self) -> str:
        return f"{self.owner}'s account: ${self.balance:.2f}"

# Usage
account = BankAccount("Alice", 100)
account.deposit(50)
account.withdraw(30)
print(account)
````

## 2. JavaScript / TypeScript Example

```ts
// TypeScript interface for a user
interface User {
    id: number;
    name: string;
    email: string;
}

// Generic function to fetch data
async function fetchData<T>(url: string): Promise<T> {
    const response = await fetch(url);
    if (!response.ok) throw new Error('Network error');
    return response.json() as Promise<T>;
}

// Example usage
fetchData<User>('https://jsonplaceholder.typicode.com/users/1')
    .then(user => console.log(user))
    .catch(err => console.error(err));
```

## 3. Bash Example

```bash
#!/usr/bin/env bash
# Backup a folder to a timestamped tar.gz

SOURCE_DIR="$HOME/projects"
BACKUP_DIR="$HOME/backups"
TIMESTAMP=$(date +%Y%m%d-%H%M%S)
FILENAME="backup-$TIMESTAMP.tar.gz"

mkdir -p "$BACKUP_DIR"
tar -czvf "$BACKUP_DIR/$FILENAME" -C "$SOURCE_DIR" .
echo "Backup saved to $BACKUP_DIR/$FILENAME"
```

## 5. SQL Example

```sql
-- Create a users table
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Insert example data
INSERT INTO users (name, email) VALUES
('Alice', 'alice@example.com'),
('Bob', 'bob@example.com');

-- Query
SELECT id, name, email FROM users WHERE created_at > NOW() - INTERVAL '1 day';
```
